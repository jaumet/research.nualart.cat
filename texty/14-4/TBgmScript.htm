GM_xmlhttpRequest({
    method: 'GET',
    url: endpoint + '?query=' + query,
    headers: {
        'User-agent': 'Mozilla/4.0 (compatible) Greasemonkey',
        'Accept': 'text/html',    
    },

    onload: function(responseDetails) {
   
    ////////////////////////////////////////////////////////////////////////////////

    }
});
